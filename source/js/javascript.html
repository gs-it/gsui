    • 						/* Syntax */
						$(selector).gsSwipe({autoplay:5000, slidesPerView:3, slidesPerGroup:3});

						/* Options information */
						$.fn.gsSwipe.option = {
							current			: 0,	// 초기화될 (slide.length/slidePerGroup) 값
							speed				: 400,	// slide 속도
							loop				: true,	// 롤링 여부
							autoplay			: 0,	// 자동실행 (사용안함:false, 사용:1000)
							simulateTouch	: false,	// PC 버젼 사용 여부
							slidesPerView	: 1,	// 화면에 보여지는 slide 수
							slidesPerGroup	: 1,	// 화면에 보여지는 slide 묶음
							onInit				: function(){}, // swipe start callback
							onChangeStart	: function(){} // slide animation complete callback
						};
